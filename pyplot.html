import matplotlib.pyplot as plt
import numpy as np

# Generate a sequence of random numbers
np.random.seed(0)  # For reproducibility
random_numbers = np.random.randint(0, 100, 5)  # Generate 5 random integers between 0 and 100

# Create figure and axes objects
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(10, 5))

# Scatter plot
ax1.scatter(range(len(random_numbers)), random_numbers, color='blue')
ax1.set_title('Scatter Plot')
ax1.set_xlabel('Index')
ax1.set_ylabel('Value')

# Plot
ax2.plot(range(len(random_numbers)), random_numbers, color='red', marker='o')
ax2.set_title('Line Plot')
ax2.set_xlabel('Index')
ax2.set_ylabel('Value')

# Show the plot
plt.tight_layout()
plt.show()
